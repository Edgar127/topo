## Usage

```jsx
import { useToasts } from '@codeday/topo/utils';

export default () => {
  const { addToast, success, error } = useToasts();
  addToast('Hello, world.');
  success('Your order was processed');
  error('Critical error', { autoDismiss: false });
  addToast('Don\'t forget to like and subscribe', { appearance: 'warning' });
}
```
