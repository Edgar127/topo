## Usage

```jsx
import { useToasts } from 'topo/utils';

export default () => {
  const { addToast } = useToasts();
  addToast('Hello, world.');
  addToast('Your order was processed', { appearance: 'success' });
  addToast('Don\'t forget to like and subscribe', { appearance: 'warning' });
  addToast('Critical error', { appearance: 'error', autoDismiss: false });
}
```
